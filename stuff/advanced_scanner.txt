# ============== UPTIME TRACKING ==============

def load_uptime_history():
    """Load uptime history from file"""
    try:
        with open(UPTIME_HISTORY, 'r') as f:
            return json.load(f)
    except:
        return {}

def save_uptime_history(history):
    """Save uptime history to file"""
    try:
        with open(UPTIME_HISTORY, 'w') as f:
            json.dump(history, f, indent=2)
    except Exception as e:
        print(f"[!] Failed to save uptime history: {e}")

def update_uptime_history(target_name, status):
    """Update uptime history for a target"""
    history = load_uptime_history()
    
    if target_name not in history:
        history[target_name] = {'checks': [], 'total_up': 0, 'total_checks': 0}
    
    # Record this check
    is_up = (status == 'UP')
    timestamp = datetime.datetime.now().isoformat()
    
    history[target_name]['checks'].append({
        'timestamp': timestamp,
        'status': status,
        'up': is_up
    })
    
    # Keep only last 96 checks (24 hours if running every 15 min)
    history[target_name]['checks'] = history[target_name]['checks'][-96:]
    
    # Calculate uptime percentage from history
    checks = history[target_name]['checks']
    if checks:
        up_count = sum(1 for c in checks if c['up'])
        uptime_percent = round((up_count / len(checks)) * 100, 1)
        history[target_name]['uptime_24h'] = uptime_percent
        history[target_name]['total_checks'] = len(checks)
    else:
        history[target_name]['uptime_24h'] = 0.0
    
    save_uptime_history(history)
    return history[target_name]['uptime_24h']

# ============== CHANGE DETECTION ==============

def load_change_history():
    """Load change detection history"""
    try:
        with open(CHANGE_HISTORY, 'r') as f:
            return json.load(f)
    except:
        return {}

def save_change_history(history):
    """Save change detection history"""
    try:
        with open(CHANGE_HISTORY, 'w') as f:
            json.dump(history, f, indent=2)
    except Exception as e:
        print(f"[!] Failed to save change history: {e}")

def detect_changes(target_name, current_hash, current_technologies):
    """Detect if target has changed"""
    history = load_change_history()
    timestamp = datetime.datetime.now().isoformat()
    
    if target_name not in history:
        # First time seeing this target
        history[target_name] = {
            'first_seen': timestamp,
            'last_hash': current_hash,
            'last_technologies': current_technologies,
            'last_changed': timestamp,
            'change_count': 0,
            'changes': []
        }
        save_change_history(history)
        return {'changed': False, 'change_type': None, 'last_changed': timestamp}
    
    target_history = history[target_name]
    previous_hash = target_history.get('last_hash')
    previous_tech = target_history.get('last_technologies', [])
    
    changes_detected = []
    
    # Check page content change
    if previous_hash and previous_hash != current_hash:
        changes_detected.append('content')
    
    # Check technology stack changes
    tech_added = set(current_technologies) - set(previous_tech)
    tech_removed = set(previous_tech) - set(current_technologies)
    
    if tech_added or tech_removed:
        changes_detected.append('technology')
    
    if changes_detected:
        # Record the change
        change_record = {
            'timestamp': timestamp,
            'change_types': changes_detected,
            'previous_hash': previous_hash[:16] + '...' if previous_hash else None,
            'new_hash': current_hash[:16] + '...',
            'tech_added': list(tech_added),
            'tech_removed': list(tech_removed)
        }
        
        target_history['changes'].append(change_record)
        target_history['changes'] = target_history['changes'][-20:]
        target_history['last_hash'] = current_hash
        target_history['last_technologies'] = current_technologies
        target_history['last_changed'] = timestamp
        target_history['change_count'] += 1
        
        save_change_history(history)
        
        print(f"    [!] CHANGE DETECTED: {', '.join(changes_detected)}")
        
        return {
            'changed': True,
            'change_type': ', '.join(changes_detected),
            'last_changed': timestamp,
            'details': change_record
        }
    else:
        # No changes, update last seen
        target_history['last_hash'] = current_hash
        target_history['last_technologies'] = current_technologies
        save_change_history(history)
        
        return {
            'changed': False,
            'change_type': None,
            'last_changed': target_history.get('last_changed', timestamp)
        }

# ============== DEEP SCAN FUNCTIONS ==============

def extract_page_title(html_content):
    """Extract page title"""
    try:
        match = re.search(r'<title>(.*?)</title>', html_content, re.IGNORECASE | re.DOTALL)
        if match:
            return match.group(1).strip()[:60]
    except:
        pass
    return "Unknown Title"

def extract_emails(content):
    """Extract email addresses"""
    email_pattern = r'\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b'
    emails = list(set(re.findall(email_pattern, content)))
    return emails[:10]

def extract_bitcoin_addresses(content):
    """Extract Bitcoin addresses"""
    btc_pattern = r'\b[13][a-km-zA-HJ-NP-Z1-9]{25,34}\b|bc1[a-z0-9]{39,59}\b'
    btc_addrs = list(set(re.findall(btc_pattern, content)))
    return btc_addrs[:5]

def extract_pgp_keys(content):
    """Extract PGP key fingerprints"""
    pgp_blocks = re.findall(r'-----BEGIN PGP PUBLIC KEY BLOCK-----(.*?)-----END PGP PUBLIC KEY BLOCK-----', 
                            content, re.DOTALL)
    
    fingerprints = []
    for block in pgp_blocks[:3]:
        key_id_match = re.search(r'[A-F0-9]{8,40}', block)
        if key_id_match:
            fingerprints.append(key_id_match.group(0))
    
    return fingerprints

def extract_onion_links(html_content, base_url):
    """Extract linked onion services"""
    onion_pattern = r'[a-z2-7]{16,56}\.onion'
    onions = list(set(re.findall(onion_pattern, html_content)))
    
    base_onion = urlparse(base_url).netloc
    onions = [o for o in onions if o != base_onion]
    
    return onions[:10]

def extract_ssh_info(headers):
    """Check for SSH disclosure"""
    ssh_header = headers.get('X-SSH-Fingerprint', '')
    if ssh_header:
        return {'disclosed': True, 'fingerprint': ssh_header}
    return {'disclosed': False}

def analyze_server_headers(headers):
    """Analyze server configuration"""
    analysis = {
        'server': headers.get('Server', 'Unknown'),
        'x_powered_by': headers.get('X-Powered-By', 'None'),
        'backend': headers.get('X-Backend-Server', 'None'),
        'real_ip_leaked': headers.get('X-Real-IP', 'No'),
        'php_version': None,
        'apache_version': None,
        'nginx_version': None
    }
    
    server = analysis['server']
    if 'Apache' in server:
        version_match = re.search(r'Apache/([\d.]+)', server)
        if version_match:
            analysis['apache_version'] = version_match.group(1)
    
    if 'nginx' in server:
        version_match = re.search(r'nginx/([\d.]+)', server)
        if version_match:
            analysis['nginx_version'] = version_match.group(1)
    
    powered = analysis['x_powered_by']
    if 'PHP' in powered:
        version_match = re.search(r'PHP/([\d.]+)', powered)
        if version_match:
            analysis['php_version'] = version_match.group(1)
    
    return analysis

def check_security_headers(headers):
    """Check security headers"""
    return {
        'csp': headers.get('Content-Security-Policy', 'Missing'),
        'x_frame_options': headers.get('X-Frame-Options', 'Missing'),
        'xss_protection': headers.get('X-XSS-Protection', 'Missing'),
        'hsts': headers.get('Strict-Transport-Security', 'Missing'),
        'referrer_policy': headers.get('Referrer-Policy', 'Missing')
    }

def extract_forms(soup):
    """Extract forms"""
    forms = []
    for form in soup.find_all('form')[:5]:
        form_data = {
            'action': form.get('action', 'N/A'),
            'method': form.get('method', 'GET').upper(),
            'inputs': []
        }
        
        for input_field in form.find_all(['input', 'textarea']):
            form_data['inputs'].append({
                'name': input_field.get('name', 'unnamed'),
                'type': input_field.get('type', 'text')
            })
        
        forms.append(form_data)
    
    return forms

def extract_javascript_files(soup, base_url):
    """Extract JavaScript files"""
    js_files = []
    for script in soup.find_all('script', src=True)[:10]:
        js_url = urljoin(base_url, script['src'])
        js_files.append(js_url)
    return js_files

def calculate_page_hash(content):
    """Calculate page hash"""
    return hashlib.sha256(content.encode()).hexdigest()

def detect_technologies(html_content, headers):
    """Detect web technologies"""
    techs = []
    
    if 'wp-content' in html_content or 'wordpress' in html_content.lower():
        techs.append('WordPress')
    if 'Drupal' in html_content:
        techs.append('Drupal')
    if 'joomla' in html_content.lower():
        techs.append('Joomla')
    if 'django' in html_content.lower():
        techs.append('Django')
    if 'Laravel' in html_content:
        techs.append('Laravel')
    if 'react' in html_content.lower():
        techs.append('React')
    if 'phpbb' in html_content.lower():
        techs.append('phpBB')
    
    if 'PHP' in headers.get('X-Powered-By', ''):
        techs.append('PHP')
    if 'ASP.NET' in headers.get('X-Powered-By', ''):
        techs.append('ASP.NET')
    
    return techs

def analyze_cookies(headers):
    """Analyze cookies"""
    set_cookie = headers.get('Set-Cookie', '')
    if not set_cookie:
        return {'present': False}
    
    return {
        'present': True,
        'secure_flag': 'Secure' in set_cookie,
        'httponly_flag': 'HttpOnly' in set_cookie,
        'samesite': 'SameSite' in set_cookie
    }

def generate_ioc_database(deep_scan_data, timestamp):
    """Generate master IOC aggregation page"""
    
    # Aggregate all IOCs
    all_emails = []
    all_btc = []
    all_pgp = []
    all_onions = []
    all_ipv4 = []
    all_domains = []
    all_file_hashes = []
    all_cves = []
    all_crypto = {'ethereum': [], 'monero': [], 'litecoin': []}
    
    for target_name, data in deep_scan_data.items():
        for email in data.get('emails', []):
            all_emails.append({'ioc': email, 'source': target_name, 'type': 'email'})
        for btc in data.get('bitcoin_addresses', []):
            all_btc.append({'ioc': btc, 'source': target_name, 'type': 'bitcoin'})
        for pgp in data.get('pgp_keys', []):
            all_pgp.append({'ioc': pgp, 'source': target_name, 'type': 'pgp'})
        for onion in data.get('linked_onions', []):
            all_onions.append({'ioc': onion, 'source': target_name, 'type': 'onion'})
        
        # Advanced IOCs
        adv_iocs = data.get('advanced_iocs', {})
        
        for ip in adv_iocs.get('ipv4_addresses', []):
            all_ipv4.append({'ioc': ip, 'source': target_name, 'type': 'ipv4'})
        
        for domain in adv_iocs.get('clearnet_domains', []):
            all_domains.append({'ioc': domain, 'source': target_name, 'type': 'domain'})
        
        hashes = adv_iocs.get('file_hashes', {})
        for md5 in hashes.get('md5', []):
            all_file_hashes.append({'ioc': md5, 'source': target_name, 'type': 'MD5'})
        for sha1 in hashes.get('sha1', []):
            all_file_hashes.append({'ioc': sha1, 'source': target_name, 'type': 'SHA1'})
        for sha256 in hashes.get('sha256', []):
            all_file_hashes.append({'ioc': sha256, 'source': target_name, 'type': 'SHA256'})
        
        for cve in adv_iocs.get('cve_ids', []):
            all_cves.append({'ioc': cve, 'source': target_name, 'type': 'cve'})
        
        crypto = adv_iocs.get('crypto_addresses', {})
        for eth in crypto.get('ethereum', []):
            all_crypto['ethereum'].append({'ioc': eth, 'source': target_name, 'type': 'ethereum'})
        for xmr in crypto.get('monero', []):
            all_crypto['monero'].append({'ioc': xmr, 'source': target_name, 'type': 'monero'})
        for ltc in crypto.get('litecoin', []):
            all_crypto['litecoin'].append({'ioc': ltc, 'source': target_name, 'type': 'litecoin'})
    
    # Save JSON version
    total_advanced = len(all_ipv4) + len(all_domains) + len(all_file_hashes) + len(all_cves)
    total_advanced += sum(len(all_crypto[k]) for k in all_crypto)
    
    ioc_json_data = {
        'generated': timestamp,
        'total_iocs': len(all_emails) + len(all_btc) + len(all_pgp) + len(all_onions) + total_advanced,
        'emails': all_emails,
        'bitcoin_addresses': all_btc,
        'pgp_keys': all_pgp,
        'linked_onions': all_onions,
        'ipv4_addresses': all_ipv4,
        'domains': all_domains,
        'file_hashes': all_file_hashes,
        'cve_ids': all_cves,
        'crypto_addresses': all_crypto
    }
    
    try:
        with open(IOC_JSON, 'w') as f:

    except Exception as e:
        print(f"[!] Failed to save IOC JSON: {e}")
    
    # Generate HTML
    html = f"""
    <!DOCTYPE html>
    <html>
    <head>
        <title>IOC Database - Master Intelligence Feed</title>
        <meta charset="UTF-8">
        <style>
            body {{ font-family: 'Courier New', monospace; background: #0d0d0d; color: #e0e0e0; padding: 20px; }}
            h1 {{ color: #00ff00; text-transform: uppercase; border-bottom: 2px solid #333; padding-bottom: 10px; }}
            h2 {{ color: #ffa500; margin-top: 30px; border-bottom: 1px solid #444; padding-bottom: 8px; }}
            .back-link {{ color: #00aaff; text-decoration: none; }}
            .back-link:hover {{ text-decoration: underline; }}
            .summary-grid {{ display: grid; grid-template-columns: repeat(auto-fit, minmax(200px, 1fr)); gap: 15px; margin: 20px 0; }}
            .summary-box {{ background: #1a1a1a; padding: 15px; border-left: 3px solid #ff3333; }}
            .summary-label {{ color: #888; font-size: 0.85em; text-transform: uppercase; }}

def check_site(target):
    """Main site checking"""
    url = target['url']
    
    try:
        start_time = time.time()
        resp = requests.get(url, proxies=PROXIES, headers=HEADERS, timeout=60, verify=False)
        latency = round(time.time() - start_time, 2)
        
        soup = BeautifulSoup(resp.text, 'html.parser')
        title = extract_page_title(resp.text)
        server = resp.headers.get('Server', 'Unknown')
        
        result = {
            'status': None,
            'status_detail': None,
            'color': None,
            'latency': f"{latency}s",
            'title': title,
            'server': server,
            'deep_data': None,
            'http_code': resp.status_code
        }
        
        if target.get('deep_scan', False):
            print(f"    [DEEP SCAN ENABLED]")
            result['deep_data'] = deep_scan_site(url, resp, soup)
        
        if 200 <= resp.status_code < 300:
            result['status'] = "UP"
            result['status_detail'] = f"HTTP {resp.status_code}"
            result['color'] = "green"
        elif resp.status_code == 403:
            result['status'] = "ISSUES"
            result['status_detail'] = "HTTP 403 Forbidden"
            result['color'] = "orange"
            result['title'] = "Access Denied (403)"
        elif resp.status_code == 401:
            result['status'] = "ISSUES"
            result['status_detail'] = "HTTP 401 Unauthorized"
            result['color'] = "orange"
            result['title'] = "Authentication Required (401)"
        elif resp.status_code == 404:
            result['status'] = "ISSUES"
            result['status_detail'] = "HTTP 404 Not Found"
            result['color'] = "orange"
            result['title'] = "Page Not Found (404)"
        elif resp.status_code >= 500:
            result['status'] = "ISSUES"
            result['status_detail'] = f"HTTP {resp.status_code} Server Error"
            result['color'] = "orange"
            result['title'] = f"Server Error ({resp.status_code})"
        else:
            result['status'] = "ISSUES"
            result['status_detail'] = f"HTTP {resp.status_code}"
            result['color'] = "orange"
            result['title'] = f"Status {resp.status_code}"
        
        return result
            
    except requests.exceptions.ConnectionError:
        return {
            'status': "DOWN",
            'status_detail': "Connection Failed",
            'color': "red",
            'latency': "-",
            'title': "Connection Failed",
            'server': "-",
            'deep_data': None,
            'http_code': None
        }
    except requests.exceptions.Timeout:
        return {
            'status': "DOWN",
            'status_detail': "Timeout (>60s)",
            'color': "red",
            'latency': ">60s",
            'title': "Timed Out",
            'server': "-",
            'deep_data': None,
            'http_code': None
        }
    except Exception as e:
        return {
            'status': "ERROR",
            'status_detail': str(e)[:40],
            'color': "red",
            'latency': "-",
            'title': str(e)[:30],
            'server': "-",
            'deep_data': None,
            'http_code': None
        }

def calculate_intelligence_summary(all_results, deep_scan_data):
    """Calculate aggregate intelligence metrics"""
    
    total_targets = len(all_results)
    up_targets = sum(1 for r in all_results if r['result']['status'] == 'UP')
    down_targets = sum(1 for r in all_results if r['result']['status'] == 'DOWN')
    issues_targets = sum(1 for r in all_results if 'ISSUES' in r['result']['status'])
    
    deep_scan_count = sum(1 for r in all_results if r['deep_scan_enabled'])
    
    # IOC counts
    total_emails = 0
    total_btc = 0
    total_pgp = 0
    total_linked_onions = 0
    
    for target_name, data in deep_scan_data.items():
        total_emails += len(data.get('emails', []))
        total_btc += len(data.get('bitcoin_addresses', []))
        total_pgp += len(data.get('pgp_keys', []))
        total_linked_onions += len(data.get('linked_onions', []))
    
    # Calculate average latency (exclude failures)
    latencies = []
    for r in all_results:
        lat = r['result']['latency']
        if lat != '-' and '>' not in lat:
            try:
                latencies.append(float(lat.replace('s', '')))
            except:
                pass
    
    avg_latency = round(sum(latencies) / len(latencies), 2) if latencies else 0
    
    # Category breakdown
    categories = {}
    for r in all_results:
        cat = r['category']
        categories[cat] = categories.get(cat, 0) + 1
    
    uptime_percentage = round((up_targets / total_targets) * 100, 1) if total_targets > 0 else 0
    
    # Count changes detected
    changes_detected = sum(1 for r in all_results if r.get('change_info', {}).get('changed', False))
    
    return {
        'total_targets': total_targets,
        'up_targets': up_targets,
        'down_targets': down_targets,
        'issues_targets': issues_targets,
        'uptime_percentage': uptime_percentage,
        'deep_scan_count': deep_scan_count,
        'total_iocs': total_emails + total_btc + total_pgp,
        'total_emails': total_emails,
        'total_btc': total_btc,
        'total_pgp': total_pgp,
        'total_linked_onions': total_linked_onions,
        'avg_latency': avg_latency,
        'categories': categories,
        'changes_detected': changes_detected
    }
